        <div id="result"></div>
    </div>
 
    <script>
        const calculateBill = (...prices) => {
            let total = 0;
            prices.forEach(p => total += p);
            const gst = amt => amt * 1.18;
            return gst(total);
        };
 
        const generateBill = () => {
            const input = prices.value.trim();
            if (!input) return result.innerText = "Please enter product prices!";
 
            const priceArray = input.split(",").map(Number);
 
            const finalAmount = calculateBill(...priceArray);
 
            result.innerText = `Total Bill (including 18% GST): â‚¹ ${finalAmount.toFixed(2)}`;
        };
    </script>

            <button onclick="showPassedStudents()">Show Passed Students</button>

        <table id="studentTable">
            <thead>
                <tr>
                    <th>Name</th>
                    <th>Roll No</th>
                    <th>Marks</th>
                </tr>
            </thead>
            <tbody>
 
            </tbody>
        </table>
 
        <h3 id="passedTitle" style="display:none;">Passed Students</h3>
        <table id="passedTable" class="passed" style="display:none;">
            <thead>

            </thead>

    <script>
        class Student {
            constructor(name, rollNo, marks) {
                this.name = name;
                this.rollNo = rollNo;
                this.marks = marks;
            }
 
            isPassed() {
                return this.marks > 40;
            }
        }
 
        const students = [];
 
        function addStudent() {
            let name = document.getElementById("name").value.trim();
            let rollNo = document.getElementById("rollNo").value;
            let marks = document.getElementById("marks").value;
 
            if(name === "" || rollNo === "" || marks === "") {
                alert("Please fill all fields!");
                return;
            }
 
            const student = new Student(name, Number(rollNo), Number(marks));
            students.push(student);
 
            document.getElementById("name").value = "";
            document.getElementById("rollNo").value = "";
            document.getElementById("marks").value = "";
 
            showAllStudents();
        }
 
        function showAllStudents() {
            const tbody = document.getElementById("studentTable").querySelector("tbody");
            tbody.innerHTML = "";
 
            students.forEach(student => {
                const row = document.createElement("tr");
                row.innerHTML = `
                    <td>${student.name}</td>
                    <td>${student.rollNo}</td>
                    <td>${student.marks}</td>
                `;
                tbody.appendChild(row);
            });
 
            document.getElementById("passedTable").style.display = "none";
            document.getElementById("passedTitle").style.display = "none";
        }
 
        function showPassedStudents() {
            const tbody = document.getElementById("passedTable").querySelector("tbody");
            tbody.innerHTML = "";
 
            const passedStudents = students.filter(s => s.isPassed());
 
            if(passedStudents.length === 0) {
                tbody.innerHTML = "<tr><td colspan='3'>No students passed yet.</td></tr>";
            } else {
                passedStudents.forEach(student => {
                    const row = document.createElement("tr");
                    row.innerHTML = `
                        <td>${student.name}</td>
                        <td>${student.rollNo}</td>
                        <td>${student.marks}</td>
                    `;
                    tbody.appendChild(row);
                });
            }
 
            document.getElementById("passedTable").style.display = "table";
            document.getElementById("passedTitle").style.display = "block";
        }
    </script>
 
</body>
</html>

const http = require('http');
 
const PORT = 3000;
 
const htmlTemplate = (title, message) => `
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>${title}</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            background: #f0f8ff;
            text-align: center;
            padding: 50px;
        }
        h1 { color: #4e73df; }
        p { font-size: 18px; color: #333; }
        a {
            display: inline-block;
            margin: 10px;
            padding: 10px 15px;
            background-color: #4e73df;
            color: white;
            text-decoration: none;
            border-radius: 5px;
        }
        a:hover { background-color: #365bc9; }
    </style>
</head>
<body>
    <h1>${title}</h1>
    <p>${message}</p>
    <div>
        <a href="/">Home</a>
        <a href="/about">About</a>
        <a href="/contact">Contact</a>
    </div>
</body>
</html>
`;
 
const server = http.createServer((req, res) => {
    let statusCode = 200;
    let title = '';
    let message = '';
 
    if(req.url === '/' || req.url === '/home') {
        title = "Welcome to My College";
        message = "This is the home page.";
    } else if(req.url === '/about') {
        title = "About Us";
        message = "Our college is dedicated to excellence in education.";
    } else if(req.url === '/contact') {
        title = "Contact Us";
        message = "Email: info@mycollege.com | Phone: +1234567890";
    } else {
        statusCode = 404;
        title = "404 Not Found";
        message = "Oops! The page you are looking for does not exist.";
    }
 
    res.writeHead(statusCode, { 'Content-Type': 'text/html' });
    res.end(htmlTemplate(title, message));
});
 
server.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}`);
});

import { useState } from 'react';
import './App.css';
 
function App() {
  const [page, setPage] = useState('home');
 
  const renderContent = () => {
    switch(page) {
      case 'home':
        return (
          <div className="content">
            <h1>Welcome to My College</h1>
            <p>Providing quality education and shaping the future.</p>
          </div>
        );
      case 'about':
        return (
          <div className="content">
            <h1>About Us</h1>
            <p>Our college is dedicated to excellence in education and student development.</p>
          </div>
        );
      case 'contact':
        return (
          <div className="content">
            <h1>Contact Us</h1>
            <p>Name: Prathamesh Shinde</p>
            <p>Email: prathameshshinde2002@gmail.com</p>
          </div>
        );
      default:
        return null;
    }
  };
 
  return (
    <div>
      <nav className="navbar">
        <h2>My College</h2>
        <div className="links">
          <button onClick={() => setPage('home')}>Home</button>
          <button onClick={() => setPage('about')}>About</button>
          <button onClick={() => setPage('contact')}>Contact</button>
        </div>
      </nav>
      {renderContent()}
    </div>
  );
}
 
export default App;
